Карты в React Native

mobile12 июля 2022

Поговорим о том как можно использовать карты в приложении на RN бесплатно. О том какой ещё может быть структура каталогов. Как можно сделать универсальный контейнер для компонентов. Чем могут быть полезны селекторы.

Структура каталогов. Как вариант - каждая фича (feature) в отдельной директории, чтобы не искать размазанные по всему проекту файлы. Общие для всех только методы API. В нашем примере это директория map.

$ tree src
src

│       ├── MapContainer.tsx

При использовании TypeScript сразу заведём типы данных в types.ts:

export interface IMarker {
  i: number
  latitude: number
  longitude: number
  active?: boolean
  visited?: boolean
}

export interface IRegion {
  latitude: number
  longitude: number
  latitudeDelta: number
  longitudeDelta: number
}

Вывод карты

Не стану углубляться в настройку хранилища, подробный рассказ об этом уже был. В исходной точке подключим само хранилище (store) и экран приложения с картой.

// App.tsx
import React from 'react'
import { Provider } from 'react-redux'
import { store } from './src/store'
import MapScreen from './src/modules/map/MapScreen'

const App = () => (
  <Provider store={store}>
    <MapScreen />
  </Provider>
)

export default App

Экран может быть скомпонован из множества элементов и контейнеров как страница на вебе. В данном случае используется лишь один контейнер с картой.

// MapScreen.tsx
import React from 'react'
import MapContainer from './MapContainer'
import MapView from './MapView'

const MapScreen = () => (
  <MapContainer>
    <MapView />
  </MapContainer>
)

export default MapScreen

А вот и первый лайфхак. Если есть необходимость использовать универсальный контейнер без привязки к конкретным компонентам (не наш случай, но всё же), можно опираться на render-props или вообще склонировать компонент, который будет передан контейнеру.

// MapContainer.tsx
import React, { ReactElement } from 'react'

type Props = {
  children: ReactElement
}

const MapContainer = ({ children }: Props) =>
  <>{React.cloneElement(children, {})}</>

export default MapContainer

Ставим библиотеку react-native-maps:

$ yarn add react-native-maps; cd ios; pod install; cd ../
  • initialRegion — координаты при инициализации карты
  • urlTemplate — для простейших нужд бесплатна карта OpenStreet
  • tracksViewChanges — должен ли маркер отслеживать изменения, для лучшей производительности можно отключить

Остальное должно быть понятно из кода. И пока обойдёмся тестовыми данными, чтобы скорее увидеть результат.

// MapView.tsx
import React from 'react'
import { StyleSheet, Dimensions } from 'react-native'
import Map, { UrlTile, Marker } from 'react-native-maps'
import { IRegion, IMarker } from './types'

const SCREEN_HEIGHT = Dimensions.get('window').height
const urlTemplate = 'https://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png'
const REGION: IRegion = {
  latitude: 59.9342802,
  longitude: 30.3350986,
  latitudeDelta: 0.5,
  longitudeDelta: 0.5,
}
const MARKERS: IMarker[] = [
  {
    i: 1421059,
    latitude: 59.903967999999999,
    longitude: 30.519534,
  },
  {
    i: 1007126,
    latitude: 59.930376000000003,
    longitude: 30.353428999999998,
  },
  {
    i: 1332648,
    latitude: 59.839900999999998,
    longitude: 30.350867000000001,
  },
]

const MapView = () => (
  <Map initialRegion={REGION} style={styles.map}>
    <UrlTile urlTemplate={urlTemplate} maximumZ={10} flipY={false} />
    {MARKERS.map((marker: IMarker) => (
      <Marker
        key={`${marker.i}-${marker.active}`}
        identifier={`${marker.i}`}
        coordinate={{ latitude: marker.latitude, longitude: marker.longitude }}
        tracksViewChanges={false}
      />
    ))}
  </Map>
)

const styles = StyleSheet.create({
  map: {
    width: '100%',
    height: SCREEN_HEIGHT,
  },
})

export default MapView

Смена маркеров при прокрутке карты

Теперь представим, что у нас есть API, где при смене региона мы получаем новые данные. А вот если просто сдвигать область на карте, данные не меняются: API отдаёт весь внушительный список маркеров, что у него есть. Выводить сразу все маркеры нельзя — скажется на производетельности. В этом случае нас выручат селекторы.

Для рассчёта ближайших маркеров понадобится библиотека: например, geolib.

// selectors.ts
import { createSelector } from 'reselect'
import { orderByDistance } from 'geolib'
import { RootState } from '../../store'

const MARKERS_LENGTH = 10

const stateSelector = (state: RootState) => state.maps
// маркеры
export const mapEntitiesSelector = createSelector(stateSelector, item => item.mapEntities)
// регион
export const mapRegionSelector = createSelector(stateSelector, item => item.region)
// отслеживание процесса загрузки данных
export const mapLoadingSelector = createSelector(stateSelector, item => item.loading)

// рассчёт ближайших маркеров на карте, отдавать 10 штук
// при изменении региона или маркеров пересчитывается автоматически
export const mapMarkersSelector = createSelector(
  [mapEntitiesSelector, mapRegionSelector],
  (data, region) => {
    if (Object.keys(region).length) {
      return orderByDistance(region, data).slice(0, MARKERS_LENGTH)
    }

    return data.slice(0, MARKERS_LENGTH)
  },
)
// store.ts
import { createAsyncThunk, createSlice, isRejected } from '@reduxjs/toolkit'
import { getMapEntities } from '../../services/api/maps'
import { IMarker, IRegion } from './types'

export const fetchMapEntities = createAsyncThunk(
  'maps/GET_MARKERS',
  async () => await getMapEntities(),
)

const LT_DEFAULT = 59.9342802
const LG_DEFAULT = 30.3350986
const REGION_DEFAULT = {
  latitude: LT_DEFAULT,
  longitude: LG_DEFAULT,
  latitudeDelta: 0.1,
  longitudeDelta: 0.1,
}

export interface IState {
  loading: boolean
  region: IRegion
  mapEntities: IMarker[]
}

const initialState: IState = {
  loading: false,
  region: REGION_DEFAULT,
  mapEntities: [],
}

const mapsSlice = createSlice({
  name: 'maps',
  initialState,
  reducers: {
    // смена локации
    changeRegion(state, action) {
      state.region = action.payload
    },
  },
  extraReducers: builder => {
    builder
      // загрузка данных
      .addCase(fetchMapEntities.pending, state => {
        state.loading = true
      })
      // получение данных
      .addCase(fetchMapEntities.fulfilled, (state, action) => {
        state.mapEntities = action.payload
        state.loading = false
      })
      .addMatcher(isRejected, state => {
        state.loading = false
      })
  },
})

export default mapsSlice
// MapContainer
import React, { useEffect, ReactElement, useCallback } from 'react'
import { useDispatch, useSelector } from 'react-redux'
import mapsSlice, { fetchMapEntities } from './store'
import { mapLoadingSelector, mapMarkersSelector, mapRegionSelector } from './selectors'
import { IRegion } from './types'

const { changeRegion } = mapsSlice.actions

type Props = {
  children: ReactElement
}

const MapContainer = ({ children }: Props) => {
  // забираем данные из хранилища
  const dispatch = useDispatch()
  const markers = useSelector(mapMarkersSelector)
  const region = useSelector(mapRegionSelector)
  const isLoading = useSelector(mapLoadingSelector)

  const handleGetMapEntities = useCallback(
    async () => await dispatch(fetchMapEntities()),
    [dispatch],
  )

  // получить от карты точное местоположение и сохранить его
  const onRegionChangeComplete = (coordinate: IRegion) => {
    dispatch(changeRegion(coordinate))
  }

  /*
    при посещении экрана дёргаем API один раз
    можно проверять есть ли уже данные
    или получать новые данные при смене локации
    это остаётся на усмотрение читателя
  */
  useEffect(() => {
    handleGetMapEntities()
  }, [handleGetMapEntities])

  // передать данные в дочерний компонент
  const CloneChildren = React.cloneElement(children, {
    markers,
    region,
    isLoading,
    onRegionChangeComplete,
  })

  return <>{CloneChildren}</>
}

export default MapContainer

Для того, чтобы уменьшить время отклика и повысить производительность при перерисовке маркеров, поставьте библиотеку throttle-debounce. Код в MapContainer изменится таким образом:

// MapContainer.tsx
import { debounce } from 'throttle-debounce'
// ...

const onRegionChangeComplete = debounce(500, (coordinate: IRegion) => {
  dispatch(changeRegion(coordinate))
})

Состояние маркеров

Не упомнишь какие из маркеров были нажаты. Здорово было бы отражать их состояние: обычный, активный или уже просмотренный. Как может выглядеть реализация?

Нам нужен новый экшен setMarkerAction для манипулирования состоянием маркера. Импортируем его из группы экшенов и отдадим дочернему компоненту.

// MapContainer.tsx
import mapsSlice, { fetchMapEntities } from './store'
const { changeRegion, setMarkerAction } = mapsSlice.actions

const MapContainer = ({ children }: Props) => {
  // ...
  const toggleMarker = (id: number | undefined) => {
    dispatch(setMarkerAction({ id }))
  }
  // ...
  const CloneChildren = React.cloneElement(children, {
    markers,
    region,
    isLoading,
    onRegionChangeComplete,
    toggleMarker,
  })
}

В дочернем компоненте определим цвета для разных состояний маркеров, а также разрешим нажимать маркер.

// MapView.tsx
// ...
interface Props {
  markers: IMarker[]
  region: IRegion
  isLoading: boolean
  onRegionChangeComplete: (region: IRegion) => void
  toggleMarker: (id: number) => void
}

const switchColor = (item: IMarker) => {
  if (item.active) return '#009a00'
  if (item.visited) return '#adbce6'
  return '#ff0040'
}

const MapView = ({
  markers,
  region,
  onRegionChangeComplete,
  toggleMarker,
  isLoading
}: Props) => {
  if (isLoading) {
    return <ActivityIndicator size="large" />
  }

  return (
    <Map
      initialRegion={region}
      style={styles.map}
      onRegionChangeComplete={onRegionChangeComplete}
      moveOnMarkerPress={false}
      removeClippedSubviews>
      <UrlTile urlTemplate={urlTemplate} maximumZ={10} flipY={false} />
      {markers.map((marker: IMarker) => {
        const { latitude, longitude, i, active } = marker
        const handlePress = () => {
          toggleMarker(i)
        }
        const color = switchColor(marker)

        return (
          <Marker
            key={`${i}-${active}`}
            identifier={`${i}`}
            coordinate={{ latitude, longitude }}
            tracksViewChanges={false}
            onPress={handlePress}
            pinColor={color}
          />
        )
      })}
    </Map>
  )
}
// ...

Наконец, добавим редьюсер, содержащий логику смены состояния. Работать это будет ровно до момента новой загрузки данных с сервера.

// store.ts
export interface IState {
  // ...
  lastActiveMarkerId: number
}

const initialState: IState = {
  // ...
  lastActiveMarkerId: -1,
}

const mapsSlice = createSlice({
  // ...
  reducers: {
    setMarkerAction(state, action) {
      const { id = -1 } = action.payload
      // поиск индекса по id
      const getMarkerIndex = (Id: number | undefined) =>
        state.mapEntities.findIndex(marker => marker.i === Id)
      let lastIndex = getMarkerIndex(state.lastActiveMarkerId)
      const currentIndex = getMarkerIndex(id)

      if (lastIndex !== -1) {
        /* снять active с предыдущего маркера */
        state.mapEntities[lastIndex].active = false
        state.mapEntities[lastIndex].visited = true
      }

      // если передан id
      if (id && currentIndex !== -1) {
        state.lastActiveMarkerId = id
        /* установить active текущему маркеру */
        state.mapEntities[currentIndex].active = true
        state.mapEntities[currentIndex].visited = false
      }
    }
  }
})

Теперь маркеры меняют цвет в зависимости от того было по ним нажатие или нет.
